function cleanup(effectFn: ReactiveEffect) {
    for (let i = 0; i < effectFn.deps.length; i++) {
        const deps = effectFn.deps[i]
        deps.delete(effectFn) //将副作用函数从依赖集合里删除
    }
    effectFn.deps.length = 0 //重置 effectFn.deps 数组
}
